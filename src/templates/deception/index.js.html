JavaScript Utility Library - index.js

This is a lightweight JavaScript utility library providing common helper functions for DOM manipulation, AJAX requests, and data processing.

Note: This library is for educational purposes only. Do not use in production without proper security review.

Library Overview

Version: 2.1.4 | Last Updated: 2023-11-15

Core Functions

// Utility Functions
const Utils = {
    // DOM Manipulation
    $(selector) { return document.querySelector(selector); },
    $$(selector) { return document.querySelectorAll(selector); },
    createElement(tag, props = {}, children = []) {
        const el = document.createElement(tag);
        Object.assign(el, props);
        children.forEach(child => {
            if (typeof child === 'string') {
                el.appendChild(document.createTextNode(child));
            } else if (child instanceof Node) {
                el.appendChild(child);
            }
        });
        return el;
    },

    // AJAX Helper
    ajax(options = {}) {
        return new Promise((resolve, reject) => {
            const xhr = new XMLHttpRequest();
            xhr.open(options.method || 'GET', options.url);
            
            if (options.headers) {
                Object.keys(options.headers).forEach(key => {
                    xhr.setRequestHeader(key, options.headers[key]);
                });
            }
            
            xhr.onload = () => {
                if (xhr.status >= 200 && xhr.status < 300) {
                    resolve(xhr.responseText);
                } else {
                    reject(new Error(`Request failed with status ${xhr.status}`));
                }
            };
            
            xhr.onerror = () => reject(new Error('Network error'));
            xhr.send(options.body || null);
        });
    },

    // Data Processing
    debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    },

    throttle(func, limit) {
        let inThrottle;
        return function(...args) {
            if (!inThrottle) {
                func.apply(this, args);
                inThrottle = true;
                setTimeout(() => inThrottle = false, limit);
            }
        };
    }
};

// Export for Node.js or browser
if (typeof module !== 'undefined' && module.exports) {
    module.exports = Utils;
} else {
    window.Utils = Utils;
}

Usage Examples

DOM Manipulation

// Select elements
const header = Utils.$('#header');
const buttons = Utils.$$('.btn');

// Create elements
const div = Utils.createElement('div', { className: 'container', id: 'main' }, [
    Utils.createElement('h1', {}, ['Hello World']),
    Utils.createElement('p', {}, ['This is a paragraph'])
]);

document.body.appendChild(div);

AJAX Request

// GET request
Utils.ajax({
    url: '/api/data',
    method: 'GET',
    headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer YOUR_TOKEN_HERE'
    }
}).then(response => {
    console.log('Data:', JSON.parse(response));
}).catch(error => {
    console.error('Error:', error);
});

// POST request
Utils.ajax({
    url: '/api/submit',
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ name: 'John', email: 'john@example.com' })
});

Security Notes

  • XSS Protection: This library does not automatically sanitize user input. Always sanitize data before inserting into DOM.
  • CSRF: AJAX requests should include CSRF tokens for state-changing operations.
  • Input Validation: Validate all inputs on both client and server side.
  • CORS: Be aware of CORS policies when making cross-origin requests.
