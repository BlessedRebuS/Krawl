serverless.yml

This is the Serverless Framework configuration for the demo application.

service: demo-app

provider:
  name: aws
  runtime: nodejs14.x
  region: us-east-1
  stage: dev
  iamRoleStatements:
    - Effect: Allow
      Action:
        - dynamodb:*
      Resource: "*"

functions:
  hello:
    handler: handler.hello
    events:
      - http:
          path: hello
          method: get
          cors: true
  process:
    handler: handler.process
    events:
      - schedule: rate(5 minutes)

resources:
  Resources:
    DemoTable:
      Type: AWS::DynamoDB::Table
      Properties:
        TableName: demo-table
        AttributeDefinitions:
          - AttributeName: id
            AttributeType: S
        KeySchema:
          - AttributeName: id
            KeyType: HASH
        BillingMode: PAY_PER_REQUEST
