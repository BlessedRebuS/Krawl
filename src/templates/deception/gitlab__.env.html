GitLab Environment Variables

Below are the contents of the .env file for the GitLab instance.

# GitLab configuration
GITLAB_HOME=/var/opt/gitlab
GITLAB_OMNIBUS_CONFIG="external_url 'https://gitlab.example.com'"
GITLAB_ROOT_PASSWORD=SuperSecret123!
GITLAB_ROOT_EMAIL=admin@example.com
SMTP_ENABLED=true
SMTP_ADDRESS=smtp.example.com
SMTP_PORT=587
SMTP_USER_NAME=gitlab_smtp
SMTP_PASSWORD=SmtpPass!@#
IMAP_ENABLED=true
IMAP_ADDRESS=imap.example.com
IMAP_PORT=993
IMAP_USER=gitlab_imap
IMAP_PASSWORD=ImapPass!@#
GITLAB_SIGNUP_ENABLED=false
GITLAB_PROJECTS_ISSUES_ENABLED=true
GITLAB_PROJECTS_MERGE_REQUESTS_ENABLED=true
GITLAB_PROJECTS_WIKI_ENABLED=true
GITLAB_BACKUP_SCHEDULE=daily
GITLAB_BACKUP_PG_SCHEMA=public
REDIS_HOST=redis.example.com
REDIS_PORT=6379
POSTGRES_HOST=db.example.com
POSTGRES_PORT=5432
POSTGRES_USER=gitlab
POSTGRES_PASSWORD=DbPass!@#
POSTGRES_DB=gitlabhq_production
