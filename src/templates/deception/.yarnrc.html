.yarnrc

This is a sample Yarn RC configuration file.

# Yarn RC file
# This is a fake configuration for demonstration purposes

# Set the registry to a custom mirror
registry "https://registry.yarnpkg.com"

# Enable offline mirror
offline-mirror "./.offline-cache"

# Set the cache folder
cache-folder "~/.yarn/cache"

# Disable progress bars
progress false

# Set the default version
version-tag-prefix "v"

# Additional settings
enableGlobalCache true
