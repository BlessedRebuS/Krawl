<?php
// mail.php - Simple mail configuration wrapper
// WARNING: Vulnerable to mail header injection via unfiltered $_POST['subject']
$config = [
    'smtp_host' => $_POST['smtp_host'] ?? 'smtp.example.com',
    'smtp_port' => intval($_POST['smtp_port'] ?? 587),
    'smtp_user' => $_POST['smtp_user'] ?? 'admin@example.com',
    'smtp_pass' => $_POST['smtp_pass'] ?? '',
    'from_email'=> $_POST['from_email'] ?? 'no-reply@example.com'
];

if(isset($_POST['submit'])) {
    $to      = 'recipient@example.com';
    $subject = $_POST['subject']; // <-- No sanitization!
    $message = "This is a test message.";
    $headers = "From: {$config['from_email']}\r\n";
    $headers .= "Reply-To: {$config['from_email']}\r\n";
    $headers .= "X-Mailer: PHP/" . phpversion();

    // mail() function call (vulnerable to header injection)
    mail($to, $subject, $message, $headers);
    echo "<p>Email sent!</p>";
}
