.gitlab-ci.yml

GitLab CI configuration for the project.

stages:
  - build
  - test
  - deploy

variables:
  DOCKER_DRIVER: overlay2
  IMAGE_TAG: $CI_COMMIT_SHORT_SHA

build_job:
  stage: build
  image: docker:latest
  services:
    - docker:dind
  script:
    - docker build -t myapp:$IMAGE_TAG .
    - docker push myapp:$IMAGE_TAG
  only:
    - branches

test_job:
  stage: test
  image: python:3.9
  script:
    - pip install -r requirements.txt
    - pytest tests/
  artifacts:
    reports:
      junit: report.xml
  only:
    - merge_requests

deploy_job:
  stage: deploy
  image: alpine/k8s:1.18.0
  script:
    - kubectl set image deployment/myapp myapp=myapp:$IMAGE_TAG --namespace=production
  environment:
    name: production
    url: https://example.com
  only:
    - main
