fetch-icanhazip.com-with python.py

This script demonstrates how to retrieve your public IP address using the icanhazip.com service.

#!/usr/bin/env python3
import urllib.request
import sys

def get_ip():
    try:
        with urllib.request.urlopen('http://icanhazip.com') as response:
            ip = response.read().decode('utf-8').strip()
            return ip
    except Exception as e:
        return f'Error: {e}'

if __name__ == '__main__':
    print(get_ip())
