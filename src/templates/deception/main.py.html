# Simple vulnerable web app (for demonstration only)
from flask import Flask, request, render_template_string

app = Flask(__name__)

@app.route('/')
def index():
    return 'Hello, World!'

@app.route('/search')
def search():
    # VULNERABLE: directly renders user input
    query = request.args.get('q', '')
    return render_template_string('
Results for: {{ query }}
', query=query)

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=8080)
